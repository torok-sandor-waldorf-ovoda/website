import { Children, isValidElement } from "react";

export function Gallery({ children, images: imagesProp, visibleCount = 4, id }) {
  const rotations = [-6, 12, -5, 9, -5, 3];

  const imagesFromChildren = (childNodes) => {
    const list = [];
    Children.forEach(childNodes, (child) => {
      if (!isValidElement(child) || child.type !== "img") return;
      const src = child.props.src;
      const alt = child.props.alt || "";
      if (src) list.push({ src, alt });
    });
    return list;
  };

  const images =
    imagesProp && imagesProp.length
      ? imagesProp
      : imagesFromChildren(children);
  if (!images.length) return null;

  const galleryId =
    id ||
    `gallery-${images[0].src.replace(/[^a-z0-9]/gi, "").slice(-12) || "default"}`;
  const thumbs = images.slice(0, visibleCount);
  const remaining = Math.max(0, images.length - visibleCount);

  return (
    <div
      class="c-gallery"
      data-gallery
      data-gallery-id={galleryId}
      data-gallery-images={JSON.stringify(images)}
      data-gallery-visible={visibleCount}
      style={{ "--thumb-count": thumbs.length }}
    >
      
        <div class="c-gallery__stack" role="group" aria-label="Galéria">
          {thumbs.map((img, i) => (
            <button
              type="button"
              key={img.src}
              class="c-gallery__thumb"
              data-index={i}
              style={{
                "--thumb-index": i,
                "--rotation": `${rotations[i % rotations.length]}deg`,
                "--delay": `${0.04 + i * 0.07}s`,
              }}
              aria-label={`${i + 1}. kép megnyitása`}
            >
              <img draggable={false} src={img.src} alt={img.alt} loading="lazy" decoding="async" />
            </button>
          ))}
        </div>
        {remaining > 0 && (
          <span class="c-gallery__badge" aria-hidden="true">
            +{remaining} fotó
          </span>
        )}  
    </div>
  );
}
